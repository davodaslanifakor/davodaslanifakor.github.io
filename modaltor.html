responsive

                    
  <vue-modaltor  
    :visible="open" 
    @hide="open = false"
    :default-width="'30%'"
    :resize-width='{1400:"40%",1200:"50%",992:"70%",768:"90%"}'
    >
        <p> 
        Lorem ipsum dolor sit amet, 
        ....
  </vue-modaltor>
                    
                  
/* additional options 
 you can handle modal size on diffrent screen sizes by editing  'resize-width' props
{992:"70%",768:"90%"} 
you can set default width by 'default-width' props (default is set to '50%')
*/
